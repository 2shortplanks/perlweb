[%- page.import({ title => 'Merging hashrefs', section => 'examples', }) -%] [% WRAPPER code_chunk %] use strict; use warnings; my $hash_ref_1 = { a => 1, b => 2, }; my $hash_ref_2 = { a => 3, c => 4, }; my $new_hash_ref = { %{$hash_ref_1}, %{$hash_ref_2} }; use Data::Dumper; print Dumper($new_hash_ref); Which produces: $VAR1 = { 'c' => 4, 'a' => 3, 'b' => 2 }; [% END %]
