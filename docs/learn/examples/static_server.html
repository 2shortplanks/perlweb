[%- page.import({ title => 'Creating a static website with Perl', section => 'examples', }) -%]

Save the following into a file called app.psgi

[% WRAPPER code_chunk %] [% INCLUDE "examples/script_head.html" -%] # Serve static files from document root with a directory index # app.psgi use Plack::App::Directory; my $app = Plack::App::Directory->new({ root => "/path/to/htdocs" })->to_app; [% END %]

Then run

[% WRAPPER code_chunk %] > plackup > Accepting connections at http://0:5000/ [% END %]
